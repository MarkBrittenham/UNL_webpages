\ctln{\bf Math 872 Problem Set 6}

\msk

Starred (*) problems are due Thursday, April 2.


\begin{description}


\item{40.} Describe how to express the Cartesian product of two simplices
$\Delta^n=[x_0,\ldots,x_n]$ and $\Delta^m=[y_0,\ldots,y_m]$
as a $\Delta$-complex, so 
that each $\Delta^n\times\{y_i\}$ and $\{x_j\}\times\Delta^m$
are simplices.

\msk

\item{(*)} 41. [Hatcher, p.131, \# 1] What familiar space is the quotient $\Delta$-complex
of a 2-simplex $[v_0,v_1,v_2]$ obtained by identifying the edges
$[v_0,v_1]$ and $[v_1,v_2]$, preserving the order of vertices?

\msk

\item{42.} Suppose that $X$ is a path-connected space, equipped with a $\Delta$–complex
structure. Show, directly from the definitions, that $H^\Delta_0(X)\cong{\mathbb Z}$. 
(You may assume without proof that the one-skeleton is connected.)

\msk

\item{43.} Compute the simplicial 
homology groups of the 3-simplex $\Delta^3$ (that is, the $\Delta$-complex
obtained from 4 vertices by gluing on 6 1-simplices, 4 2-simplices and a 
single 3-simplex in the ``obvious'' way). 

\msk

\item{(*)} 44. [Hatcher, p.131, \# 7] Find a way of identifying the faces of a 3-simplex
$\Delta^3$ to produce a $\Delta$-complex structure on $S^3$ having a single 3-simplex, 
and compute the simplicial homology groups of this $\Delta$-complex.

\msk

\item{45.} Compute the simplicial homology groups of the two-sphere, directly
from the definitions, using the $Delta$–complex structure coming from the boundary of a
tetrahedron (i.e., 3-simplex). 

\msk

\item{(*)} 46. In the group $G = {\mathbb Z}\times{\mathbb Z}$, 
consider the subgroup $H$ generated by $(-5,1)$ and $(1,-5)$. Show that 
the quotient group $G/H$ is cyclic.
Which of the standard cyclic groups is it isomorphic to?

\msk

\item{47.} Suppose that $A$ and $B$ are subgroups of the abelian group $C$. We define
$A + B = \{a + b\ :\ a\in A, b\in B\}\subseteq C$.

(1) Show that $A + B$ is a subgroup of $C$.

(2) There is a natural homomorphism $\phi: A\times B \rightarrow A + B$, 
defined by $\phi(a, b) = a + b$. Show that $\phi$ is surjective,
and show that $\phi$ is an isomorphism if and only if $A\cap B = \{0\}$.



\end{description}
